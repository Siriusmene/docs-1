```csharp
// Basic user
var user = new StatsigUser("user-123");

// User with custom attributes
var userWithAttributes = new StatsigUser("user-123")
{
    Email = "user@example.com",
    Country = "US",
    CustomAttributes = new Dictionary<string, object>
    {
        ["subscription_tier"] = "premium",

    }
};

// User with custom IDs for group-level experiments

{
    CustomIds = new Dictionary<string, string>
    {
        ["companyId"] = "company-456",
        ["teamId"] = "team-789"
    }
};
```
