At the root of your component tree, wrap your app in a [`StatsigProvider`](#statsig-provider) and set `waitForInitialization=true`.

```jsx
import { useState } from "react";
import { StatsigProvider } from "statsig-react";


function App() {
  const [user, setUser] = useState({userID: "a-user-id"});

  return (
    <StatsigProvider
      sdkKey="<STATSIG_CLIENT_SDK_KEY>"
      user={user}
      // StatsigOptions (Not Required)
      options={{
        environment: { tier: "staging" },
      }}
      waitForInitialization={true} // Recommended to ensure latest values a used
    >
      <div className="App">{/* Rest of App ... */}</div>
    </StatsigProvider>
  );
}
```

:::note
If you do not use `waitForInitialization`, all checks to Statsig will return default values until initialization completes.
:::
